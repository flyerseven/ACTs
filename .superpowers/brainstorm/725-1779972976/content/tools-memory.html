3. tools.py — 工具注册与调用

class ToolRegistry:
    """工具注册中心 — 支持动态增删、参数验证、超时重试"""

    # -- 注册 --
    register(tool: ToolDef) → None
    register_from_func(func: Callable, **meta) → None  # 自动从签名+docstring推断
    register_from_openai(schema: dict, func: Callable) → None
    unregister(name: str) → None
    list_tools() → list[ToolDef]
    list_openai_schemas() → list[dict]  # 生成 OpenAI function calling 格式

    # -- 调用 --
    async call(name: str, arguments: dict) → ToolCall
        # 1. 查找工具 → 2. 验证参数 → 3. 超时包装 → 4. 执行 → 5. 重试(需配置)

    # -- 参数推断 --
    _infer_schema_from_func(func: Callable) → dict
        # 类型注解 → JSON Schema: str→string, int→integer, float→number
        # docstring 首行 → description, :param xxx: → 参数描述

# 使用示例:
registry = ToolRegistry()
registry.register_from_func(
    lambda query: search_web(query),
    name="web_search",
    description="搜索互联网获取最新信息"
)
result: ToolCall = await registry.call("web_search", {"query": "Python 3.12"})

4. memory.py — 记忆管理器

class MemoryManager:
    """管理会话内的消息列表 + 自动摘要压缩"""

    # -- 消息管理 --
    add(role, content, **meta) → None
    get_messages() → list[Message]  # 所有消息
    get_context_messages(max_tokens: int) → list[dict]
        # 智能截取：system prompt 一定保留，recent messages 优先

    # -- 摘要压缩 --
    async compress(llm_adapter, trigger_tokens: int = 4000)
        # 当消息估算 token 超过阈值时触发
        # 1. 取最早 50% 的消息
        # 2. 让 LLM 生成摘要
        # 3. 删除旧消息，在 system prompt 后插入摘要

    # -- Token 估算 --
    estimate_tokens() → int  # 简单估算: char_count / 4

    # -- 快照 --
    to_dict() → dict
    from_dict(data: dict) → MemoryManager

摘要压缩触发时机

条件 动作
estimate_tokens() > 6000 自动压缩到 ~3000 tokens 以内
每 15 个 assistant 回合 对前 10 轮生成摘要（轻量压缩）
手动调用 compress() 立即压缩，force=True 强制全量摘要
