8. safety.py — 安全检查

class SafetyChecker:
    """安全检查 — 内置默认策略 + 可注册自定义钩子"""

    max_steps: int = 50
    tool_whitelist: set[str] | None = None  # None = 全部允许
    confirm_sensitive: set[str] = {"exec", "shell", "file_delete"}
    stop_requested: bool = False

    # -- 核心检查 --
    should_stop(state: AgentState) → bool
        # 1. max_steps 超限? → emit("hit_max_steps")
        # 2. stop_requested? → emit("user_interrupt")
        # 3. 同错误连续5次? → emit("error_loop")

    check_tool(tool_name: str, arguments: dict) → bool
        # 1. 白名单过滤
        # 2. 敏感操作标记 → 外部可决定是否确认

    request_stop() → None  # 紧急停止

    # -- 钩子系统 --
    before_action(callback) → None
    after_action(callback) → None
    _run_hooks(hook_name: str, context: dict) → bool  # False = 阻止

9. llm.py — LLM 适配器

class LLMResponse(BaseModel):
    content: str
    tool_calls: list[ToolCallRequest] = []
    usage: dict = {}

class LLMAdapter(ABC):
    """抽象接口 — 用户可实现此接口接入任何 LLM"""
    async chat(messages: list[dict], tools: list[dict] | None) → LLMResponse
    async chat_stream(messages, tools) → AsyncGenerator[str]

class OpenAIAdapter(LLMAdapter):
    """内置的 OpenAI 兼容适配器 — httpx + SSE streaming"""
    base_url: str = "https://api.openai.com/v1"
    api_key: str
    model: str = "gpt-4o"
    max_retries: int = 3
    timeout: float = 60.0

class CallbackAdapter(LLMAdapter):
    """自定义 chat 回调 — 用户只需传入 async 函数"""
    _chat_fn: Callable  # async def fn(messages, tools) -> str
