5. reflect.py — 反思与自我修正

class Reflector:
    """每 N 步自动反思，检测偏离、循环、重复"""

    reflect_interval: int = 3  # 每3步反思一次

    async reflect(state: AgentState, memory: MemoryManager, llm) → Reflection

    # -- 检测方法（不依赖 LLM，纯规则）--
    detect_repetition(steps: list[Step]) → bool
        # 连续3步调用相同工具+相同参数 → 判定循环
    detect_off_track(goal, steps) → float
        # 简单关键词匹配，评估与目标的偏离度 0~1
    summarize_errors(errors: list[str]) → str
        # 去重 + 类型归纳

class Reflection(BaseModel):
    """反思结果"""
    should_continue: bool
    strategy_adjustment: str = ""  # LLM 给出的策略调整建议
    is_stuck: bool = False
    detected_loop: bool = False
    off_track_score: float = 0.0
    summary: str = ""

6. observe.py — 观测系统

class Observer:
    """统一的可观测性层：日志 + 回调 + Mermaid + 指标"""

    # -- 事件发射 --
    on_event(callback: Callable[[Event], None]) → None
    emit(event: Event) → None  # 触发所有注册的回调

    # -- Mermaid 生成 --
    generate_mermaid(state: AgentState) → str
        # 输出 flowchart TD，展示每步的决策路径

    # -- 指标收集 --
    get_report() → MetricsReport
        # 执行时间 / 总步数 / 工具调用次数 / 成功率 / Token 消耗

class Event(BaseModel):
    type: Literal["step_start","step_end","tool_call","tool_result",
                  "reflection","error","paused","stopped","done"]
    timestamp: datetime
    data: dict = {}

7. engine.py — 决策循环（核心）

DecisionLoop.run() 主流程
class AgentEngine:
    """一切组件的组装点 — 用户只跟这个类交互"""

    state: StateManager
    tools: ToolRegistry
    memory: MemoryManager
    reflector: Reflector
    observer: Observer
    safety: SafetyChecker
    llm: LLMAdapter

    async run(goal: str, **kwargs) → AgentState:
        state.start(goal)
        observer.emit(Event(type="start"))

        while state.status == "running":
            if safety.should_stop(state): break

            # 1. OBSERVE — 收集上下文
            context = memory.get_context_messages()
            step = Step(index=state.current_step_index, phase="observe")

            # 2. THINK — LLM 推理下一步行动
            step.phase = "think"
            response = await llm.chat(context, tools.list_openai_schemas())
            step.thought = response.content

            # 3. ACT — 执行工具调用（如果有）
            if response.tool_calls:
                for tc in response.tool_calls:
                    if not safety.check_tool(tc.name): continue
                    step.phase = "act"
                    step.tool_call = await tools.call(tc.name, tc.arguments)
                    memory.add("tool", str(step.tool_call.result), name=tc.name)

            # 4. REFLECT — 每 N 步反思
            if state.current_step_index % reflector.reflect_interval == 0:
                step.phase = "reflect"
                reflection = await reflector.reflect(state, memory, llm)
                if reflection.is_stuck: state.stop("检测到死循环")

            state.add_step(step)
            observer.emit(Event(type="step_end", data=step))

        return state
