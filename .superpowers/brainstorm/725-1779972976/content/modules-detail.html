模块详细设计 — 类型系统 & 状态管理

Pydantic 模型定义所有核心数据结构

1. types.py — 共享类型定义

class ToolDef(BaseModel):
    """工具定义 — 同时支持 OpenAI 格式和 Python 函数"""
    name: str
    description: str
    parameters: dict  # JSON Schema
    func: Callable | None = None  # Python 函数引用（可选）
    is_async: bool = False
    timeout_sec: float = 30.0
    max_retries: int = 2

class ToolCall(BaseModel):
    """工具调用记录"""
    id: str
    tool_name: str
    arguments: dict
    result: Any = None
    error: str | None = None
    started_at: datetime
    finished_at: datetime | None = None
    duration_ms: float = 0.0
    retry_count: int = 0

class Step(BaseModel):
    """决策循环中的单步"""
    index: int
    phase: Literal["observe", "think", "act", "reflect"]
    thought: str = ""  # LLM 的推理输出
    tool_call: ToolCall | None = None
    observation: str = ""  # 环境反馈
    reflection: str = ""
    is_completed: bool = False

class AgentState(BaseModel):
    """Agent 的全局运行状态"""
    goal: str
    sub_goals: list[str] = []
    steps: list[Step] = []
    current_step_index: int = 0
    status: Literal["idle","running","paused","done","failed","stopped"] = "idle"
    errors: list[str] = []
    metrics: dict = {}  # steps, tokens, tool_calls, elapsed
    started_at: datetime | None = None

class Message(BaseModel):
    """对话消息"""
    role: Literal["system", "user", "assistant", "tool"]
    content: str
    tool_call_id: str | None = None
    name: str | None = None  # tool name for tool role

2. state.py — 状态管理器

class StateManager:
    """管理 Agent 运行时的全部状态，纯数据操作，无副作用"""

    # -- 生命周期 --
    start(goal: str) → None
    pause() → None
    resume() → None
    stop(reason: str) → None

    # -- 步骤跟踪 --
    add_step(step: Step) → None
    update_current_step(**kwargs) → None
    get_last_n_steps(n: int) → list[Step]

    # -- 子目标 --
    set_sub_goals(goals: list[str]) → None
    complete_sub_goal(index: int) → None

    # -- 错误 --
    record_error(error: str) → None
    get_error_summary() → str  # 去重 + 归纳

    # -- 指标 --
    update_metrics(**kwargs) → None
    get_metrics() → dict

    # -- 状态快照 --
    to_dict() → dict
    from_dict(data: dict) → StateManager
